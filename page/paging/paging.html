еҗҺз«ҜжҹҘиҜўйҮҸжҜ”иҫғеӨ§зҡ„йЎөйқўйңҖиҰҒж”№йҖ дёәеҗҺз«ҜеҲҶйЎөпјҢеӨ§дҪ“йңҖиҰҒжңүдёүдёӘең°ж–№ж”№йҖ дёҖдёӢгҖӮеҗҺйқўзҡ„дёӨдёӘйғЁеҲҶпјҢз®—жҳҜдёҖеҠіж°ёйҖёзҡ„еӨҚеҲ¶дёҖж¬ЎпјҢд»ҘеҗҺжӢ·иҙқе°ұиЎҢдәҶ

  • 1.еҗҺз«ҜжҸҗдҫӣж•°жҚ®зҡ„processзұ»ж–№жі•ж”№йҖ 
  • 2.жӢ·иҙқеҲҶйЎөPageModelAndviewзұ»еҲ°йЎ№зӣ®дёӯ
  • 3.жӢ·иҙқеҲҶйЎөhtmlзҡ„jspж–Үд»¶еҲ°йЎ№зӣ®дёӯ

        ModelAndView mv = new PageModelAndview("admin/xxx", p, count, psize);

2.PageModelAndviewзұ»

йЎ№зӣ®дёӯж–°е»әдёҖдёӘ PageModelAndview зұ»пјҢз„¶еҗҺе°ҶдёӢйқўд»Јз ҒжӢ·иҙқиҝҮеҺ»еҚіеҸҜпјҲжӯӨзұ»еҸҜд»Ҙж”ҫеңЁеҹәзЎҖзұ»ж–Үд»¶еӨ№дёӢпјҢж–№дҫҝз®ЎзҗҶпјү

                    
        package com.yuncheng.zyuser.modelandview;

        import java.io.IOException;
        import org.springframework.web.servlet.ModelAndView;

        /**
         *
         * @desc еҲҶйЎөmodelandview
         * @param p еҪ“еүҚеҲҶйЎө
         * @param allsize жҖ»жқЎж•°
         * @param pagesize еҚ•йЎөй•ҝеәҰ
         */
        public class PageModelAndview extends ModelAndView {

            public PageModelAndview(String view, int p, long allsize, long pagesize) throws IOException {

                super(view);

                long showpage = 5;
                long begin = 1;// иө·е§ӢйЎөйқў
                long allpage = (allsize / pagesize);//жҖ»йЎөж•°
                long end = 1; // з»“жқҹйЎөйқў
                boolean before = false;
                boolean next = false;
                boolean haspage = true;
                //и®Ўз®—иө·е§ӢйЎө,з»“жқҹйЎө
                if (allpage <= showpage) {
                    begin = 1;
                    end = allpage;
                } else if (p <= (showpage / 2 + 1)) {
                    begin = 1;
                    end = showpage;
                    next = true;
                } else if (p <= (allpage - (showpage / 2))) {
                    before = true;
                    next = true;
                    begin = p - (showpage / 2);
                    end = p + (showpage / 2);
                } else {
                    before = true;
                    begin = allpage - showpage + 1;
                    end = allpage;
                }
                if (p != 1 && allpage > 1) {
                    before = true;
                }
                if (allpage > 1 && p != allpage) {
                    next = true;
                }
                if (allpage < p) {
                    haspage = false;
                }
                super.addObject("haspage", haspage);
                super.addObject("cpage", p);
                super.addObject("pageBegin", begin);
                super.addObject("pageEnd", end);
                super.addObject("pageNext", next);
                super.addObject("pageBefore", before);
            }
        }
